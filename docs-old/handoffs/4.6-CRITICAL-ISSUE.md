## ⚠️ PROBLEM SUMMARY

Story 4.6 artifacts (Visual Composition Guide, Badge Prompt Section, Story Draft) specify **hardcoded badge colors, typography, and positioning** that ignore the existing **Visual Signature System** (Fase 1 complete).

**Impact:** All stores will have identical badges (red promotional, orange urgency, Inter font) regardless of their brand identity (`primary_color`, `store_name_typography`).

**This defeats the purpose of Visual Signature personalization and reintroduces the "generic" problem we're trying to solve.**

**Visual Composition Guide specifies:**
```

urgency → Background: #FF8C00 (orange) — ALWAYS

**Contrast calculation:**
```typescript
function getTextColor(backgroundColor: string): string {
  const luminance = calculateRelativeLuminance(backgroundColor);
  return luminance > 0.5 ? '#1F2937' : '#FFFFFF';
}

function calculateRelativeLuminance(hex: string): number {
  const rgb = hexToRgb(hex);
  const [r, g, b] = rgb.map(c => {
    c = c / 255;
    return c <= 0.03928 ? c / 12.92 : Math.pow((c + 0.055) / 1.055, 2.4);
  });
  return 0.2126 * r + 0.7152 * g + 0.0722 * b;
}
```

// lib/ai/renderer/canvas.ts — NEW rendering logic

function renderBadgeWithVisualSignature(
  ctx: CanvasRenderingContext2D,
  badgeConfig: BadgeConfig,
  visualSignature: VisualSignature,
  context: VisualContext
): void {
  // 1. Color from Visual Signature (NOT hardcoded)
  const backgroundColor = visualSignature.primary_color;
  const opacity = context === 'premium' ? 0.90 : 0.95;
  
  // 2. Text color via contrast calculation (WCAG AA)
  const textColor = getContrastTextColor(backgroundColor);
  
  // 3. Typography from Visual Signature (NOT hardcoded)
  const fontFamily = visualSignature.store_name_typography.font_family || 'Inter';
  const baseFontWeight = visualSignature.store_name_typography.font_weight || 700;
  const contextAdjustment = (context === 'promotional' || context === 'urgency') ? 100 : 0;
  const fontWeight = Math.min(900, Math.max(400, baseFontWeight + contextAdjustment));
  
  // 4. Position from Motor 2 suggestion + overlap adjustment
  const suggestedPosition = badgeConfig.badgePosition; // 'top-left', 'top-center', 'top-right'
  const finalPosition = adjustPositionForOverlap(suggestedPosition, productArea, canvasSize);
  
  // 5. Render badge
  ctx.fillStyle = `${backgroundColor}${Math.round(opacity * 255).toString(16)}`;
  ctx.fillRect(finalPosition.x, finalPosition.y, badgeWidth, badgeHeight);
  
  ctx.fillStyle = textColor;
  ctx.font = `${fontWeight} ${fontSize}px ${fontFamily}`;
  ctx.fillText(badgeConfig.badgeText, finalPosition.x + padding, finalPosition.y + badgeHeight / 2);
}

function getContrastTextColor(backgroundColor: string): string {
  const luminance = calculateRelativeLuminance(backgroundColor);
  return luminance > 0.5 ? '#1F2937' : '#FFFFFF';
}
```
